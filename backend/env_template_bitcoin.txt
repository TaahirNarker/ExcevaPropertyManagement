# ==============================================
# BITCOIN LIGHTNING PAYMENT CONFIGURATION
# ==============================================
# Add these variables to your .env file for Bitcoin payment functionality

# Strike API Configuration (Required for Bitcoin payments)
STRIKE_API_BASE_URL=https://api.strike.me/v1
STRIKE_API_KEY=your-strike-api-key-here
STRIKE_WEBHOOK_SECRET=your-strike-webhook-secret-here

# Payment Configuration
LIGHTNING_INVOICE_EXPIRY_MINUTES=15
PAYMENT_BASE_URL=http://localhost:3000/pay

# Admin Notification Emails (comma-separated)
PAYMENT_NOTIFICATION_EMAILS=admin@yourapp.com,manager@yourapp.com
ADMIN_BASE_URL=http://localhost:8000

# ==============================================
# STRIKE API SETUP INSTRUCTIONS
# ==============================================
#
# 1. Create a Strike account at https://strike.me/
# 2. Navigate to the Strike API documentation
# 3. Generate your API key from the Strike dashboard
# 4. Set up webhook endpoints for payment notifications
# 5. Copy your API credentials to the variables above
# 6. Configure webhook URL: http://yourdomain.com/api/payments/webhook/strike/
#
# Required Webhook Events:
# - invoice.created
# - invoice.updated  
# - invoice.paid
# - invoice.canceled
